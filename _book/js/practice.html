function each(obj,callback){
    if(typeof(obj)!=="object") return false;
    // var newobj = obj instanceof Array? []: {};
    for(let key in obj){
        if(callback.call(obj[key],key,obj[key])===false){
            break;
        }
        // console.log(key);
        // console.log(obj[key]);
        // callback.call(obj[key],key,obj[key]);
    }
    return obj;
}
